#### Impact

**At production scale (1000 API requests/day):**
- If 5% of requests have invalid positionRank (common with missing data): **50 calculations/day silently wrong**
- Users see incorrect TEP values, trust erodes
- Support tickets: "Why is my TE1 showing 1.0x multiplier?"
- Debugging time: **30 min per ticket × 5 tickets/week = 2.5 hours/week wasted**

**Current state:**

**After fix:**

- Error message guides toward correct usage

#### Fix

Add validation that returns error object instead of silent default:

```javascript
function getScarcityMultiplier(positionRank) {
  // Validate input
  if (positionRank === null || positionRank === undefined) {
    throw new Error(
      'Position rank is required. Provide a valid rank (1-999) or use calculateBatchTEPValues() to auto-rank TEs.'
    );
  }
  
  if (typeof positionRank !== 'number' || !Number.isFinite(positionRank)) {
    throw new Error(
      `Position rank must be a finite number. Got: ${typeof positionRank} (${positionRank}). Example: positionRank: 5`
    );
  }
  
  if (positionRank < 1) {
    throw new Error(
      `Position rank must be >= 1 (TE1 = best). Got: ${positionRank}. Use null/undefined for unranked players, not 0.`
    );
  }
  
  const rank = Math.floor(positionRank);
  
  for (const tier of SCARCITY_TIERS) {
    if (rank >= tier.minRank && rank <= tier.maxRank) {
      const rangeSize = tier.maxRank - tier.minRank + 1;
      const posInRange = rank - tier.minRank;
      const decay = (posInRange / rangeSize) * INTRA_TIER_DECAY;
      return Math.max(Number((tier.multiplier - decay).toFixed(4)), 1.00);
    }
  }
  
  return 1.00; // Waiver tier (TE25+)
}
```

**What changed:**
- Null/undefined throws error with guidance to use batch function
- Non-number throws error showing actual type received + example
- Negative rank throws error explaining rank numbering + correct usage
- Each error includes recovery steps and correct usage example

function getAgeCurveModifier(age) {
  // Null/undefined is acceptable (player age unknown)
  if (age === null || age === undefined) {
    return 1.00; // Neutral modifier for unknown age
  }
  
  // Validate type
  if (typeof age !== 'number') {
    throw new Error(
      `Age must be a number. Got: ${typeof age} (${age}). Use Number(age) to convert strings, or pass null for unknown age.`
    );
  }
  
  // Validate range
  if (!Number.isFinite(age) || age < 0) {
    throw new Error(
      `Age must be a non-negative finite number. Got: ${age}. Valid range: 0-99, or null for unknown.`
    );
  }
  
  // Validate reasonableness (optional but helpful)
  if (age > 50) {
    console.warn(`Warning: Age ${age} is unusually high for NFL player. Applying max age curve discount.`);
  }
  
  for (const band of AGE_CURVE) {
    if (age >= band.minAge && age <= band.maxAge) {
      return band.modifier;
    }
  }
  return 1.00;
}
```

**What changed:**
- Null/undefined explicitly allowed (documented as "unknown age")
- Type mismatch throws error with conversion guidance
- Invalid range throws error with valid range guidance
- Warning for suspicious ages (catches data errors early)

Add `validationError` field to return object when input is invalid:

function calculateTEPValue({
  baseValue,
  position,
  positionRank = null,
  age = null,
  advancedStats = {},
  tepLevel = 'tep3',
}) {
  // Validate inputs
  if (baseValue === null || baseValue === undefined) {
    return {
      tepValue: 0,
      isAdjusted: false,
      multiplierBreakdown: null,
      tierInfo: { tier: 11, label: 'Waiver Wire' },
      validationError: {
        field: 'baseValue',
        message: 'Base value is required. Provide a number between 0-10000.',
        receivedValue: baseValue,
        receivedType: typeof baseValue,
      },
    };
  }
  
  if (!Number.isFinite(baseValue)) {
    return {
      tepValue: 0,
      isAdjusted: false,
      multiplierBreakdown: null,
      tierInfo: { tier: 11, label: 'Waiver Wire' },
      validationError: {
        field: 'baseValue',
        message: 'Base value must be a finite number (not NaN or Infinity).',
        receivedValue: baseValue,
        receivedType: typeof baseValue,
        suggestion: 'Check your data source. Use 0 for players with no value, not NaN.',
      },
    };
  }
  
  if (baseValue < 0) {
    return {
      tepValue: 0,
      isAdjusted: false,
      multiplierBreakdown: null,
      tierInfo: { tier: 11, label: 'Waiver Wire' },
      validationError: {
        field: 'baseValue',
        message: 'Base value cannot be negative.',
        receivedValue: baseValue,
        suggestion: 'Player values are 0-10000. Use 0 for worthless players.',
      },
    };
  }
  
  // ... rest of function
}
```

**What changed:**

- Includes: field name, error message, received value/type, suggestion
- Non-breaking: existing code ignores new field
- New code can check for validationError before using tepValue

```javascript
function calculateAdvancedStatBonus(stats) {
  if (!stats || typeof stats !== 'object') return 1.00;
  
  // Valid stat keys
  const VALID_KEYS = ['targetShare', 'routeParticipation', 'rzTargetShare', 'receptionsPerGame'];
  
  // Check for unknown keys (potential typos)
  const providedKeys = Object.keys(stats);
  const unknownKeys = providedKeys.filter(k => !VALID_KEYS.includes(k));
  
  if (unknownKeys.length > 0) {
    console.warn(
      `[TEP] Unknown advanced stat keys: ${unknownKeys.join(', ')}. ` +
      `Valid keys: ${VALID_KEYS.join(', ')}. ` +
      `Check for typos in your stat object.`
    );
  }
  
  let bonus = 0;
  
  // Target Share: 0-5% bonus
  const ts = stats.targetShare;
  if (typeof ts === 'number' && ts > 0) {
    const { ELITE, HIGH, MID } = ADVANCED_STATS.TARGET_SHARE;
    if (ts >= ELITE.threshold) bonus += ELITE.bonus;
    else if (ts >= HIGH.threshold) bonus += HIGH.bonus;
    else if (ts >= MID.threshold) bonus += MID.bonus;
  }
  
  // [rest of function unchanged]
  
  // Cap total bonus
  return Number((1.00 + Math.min(bonus, ADVANCED_STATS.MAX_BONUS)).toFixed(4));
}
```

**What changed:**
- Added VALID_KEYS array documenting expected keys
- Checks for unknown keys on each call
- Warns developer if typo detected
- Suggests correct key names
- Non-breaking: only adds warning, doesn't throw
